//定义了几个常量，代表Arduino板上的引脚编号
constintLED2=13;//LED连接到的引脚，通常是Arduino板上的内置LED
constintsoundSensorPin=A0;//声音传感器连接到的模拟输入引脚
constintdigitalPin=7;//另一个数字传感器或开关连接到的数字输入引脚
constintthresholdSound=510;//设定声音传感器读数的阈值
//定义了几个整型变量，用于存储传感器读数和中间值
intsoundVal=0;//存储声音传感器的读数
intdigitalVal=0;//存储数字引脚的读数
voidsetup(){
pinMode(LED2,OUTPUT);//设置LED2引脚为输出模式
pinMode(soundSensorPin,INPUT);//设置声音传感器引脚为输入模式
pinMode(digitalPin,INPUT);//设置数字引脚为输入模式
Serial.begin(9600);//初始化串口通信，波特率设置为9600
}
voidloop(){
soundVal=analogRead(soundSensorPin);//读取声音传感器的模拟值
digitalVal=digitalRead(digitalPin);//读取数字引脚的电平值
//通过串口输出读取到的声音传感器和数字引脚的值
Serial.print("SoundValue:");
Serial.print(soundVal);
Serial.print("|DigitalPinValue:");
Serial.println(digitalVal);
//判断声音传感器的读数是否大于设定的阈值，或者数字引脚是否为高电平
if(soundVal>thresholdSound||digitalVal==HIGH){
digitalWrite(LED2,HIGH);//如果满足上述条件之一，则点亮LED
Serial.println("1");//同时通过串口输出"1"
}
//判断数字引脚是否为低电平
elseif(digitalVal==LOW){
digitalWrite(LED2,LOW);//如果是，则熄灭LED
Serial.println("LEDOff");//同时通过串口输出"LEDOff"
delay(10);//延时10毫秒
}
//其他情况（即声音传感器的读数没有超过阈值，且数字引脚不是高电平也不是低电平）
else{
digitalWrite(LED2,HIGH);//点亮LED
delay(10);//延时10毫秒
}
}
